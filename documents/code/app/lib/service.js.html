Service.prototype.getDetailsFromZip = function(path, cb) {
var that = this;
this.zipper.getFilesInfo(path, function(err, data) {
 if (err) {
   e = new AppError(that.getStatusCode("badRequest"),
     "Invalid or unsupported zip format.", err);
   return cb(e);
 }
 return cb(null, data);
});
};
