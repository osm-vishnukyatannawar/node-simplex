DevicesService.prototype.normalizeInsertData = function(deviceDataObj, values) {
  deviceDataObj['tagCurrentThreshRecId'] = uuid.v4();
  deviceDataObj['tagPimConfigRecId'] = uuid.v4();
  deviceDataObj['deviceTypeRecId'] = uuid.v4();
  deviceDataObj['deviceId'] = deviceDataObj['manufacturerName'] + deviceDataObj['modelNumber'];
  deviceDataObj['createdBy'] = values.userId;
  deviceDataObj['createdOn'] = utility.toMySQLDateTime();
  return deviceDataObj;
};

DevicesService.prototype.getDevicesForCsv = function(clientDataObj , cb) {
  var that = this;
  async.waterfall([
    function(wfcb) {
      that.mDevices.getAllDevicesForCsv(wfcb);
    },
    function(data, wfcb) {
      that.getArrayForCsv(data, wfcb);
    },
    function(array, wfcb) {
      var csvHelp = new csv();
      csvHelp.readObjCreateCsv(array, wfcb);
    },
    function(stringToWrite, wfcb) {
      that.writeToFile(stringToWrite, clientDataObj['userId']+global.__CONFIG__.calibration.allRecordsCsvFileName, wfcb);
    }
  ], function(err, data) {
    return cb(err, data);
  });

};

DevicesService.prototype.writeToFile = function(data, fileName, cb) {
  fs.writeFile(global.__CONFIG__.app_base_path + "../" + global.__CONFIG__.filesFolderName + "/" + fileName,
    data,
    function(err) {
      if (err) {
        return cb(err);
      } else {
        return cb(null);
      }
    });

};

DevicesService.prototype.getArrayForCsv = function(data, cb) {
  var baseArray = [];
  var labelsArray = [];
  for (var index in data) {
    var valuesArray = [];
    var element = data[index];
    if (labelsArray.length == 0) {
      for (var key in element) {
        labelsArray.push(key);
      }
      baseArray.push(labelsArray);
    }
    for (var key in element) {
      valuesArray.push(element[key]);
    }
    baseArray.push(valuesArray);
  }
  cb(null, baseArray);
};
