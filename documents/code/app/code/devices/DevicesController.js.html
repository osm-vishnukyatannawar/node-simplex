var util = require("util");

function DevicesController(app) {
  Controller.call(this);
  this.sDevice = new DeviceService();
  app.get(__CONFIG__.app_base_url_token + 'devices/:id', this.getDevices.bind(this));
  app.httpPost('upload-csv/', this.uploadCsv.bind(this), false, true);
  app.httpGet('export-devices/', this.exportDeviceCsv.bind(this), false, true);
  app.httpGet('unexported-rows/', this.unExportedCsvRows.bind(this), false, true);
  app.httpGet('calibration/details/', this.getCalibrationDetails.bind(this), false, true);

}

util.inherits(DevicesController, Controller);

DevicesController.prototype.getDevices = function(request, response) {

}

DevicesController.prototype.uploadCsv = function(request, response) {
  var that = this;
  var path = request.files.uploadFile.path;
  var userId = request.currToken.userID;
  this.sDevice.srvcGetDataForCalibrations({
    'path': path,
    'userId': userId
  }, function(err, data) {
    logger.logAppInfo(JSON.stringify(data));
    that.sendResponse(err, data, response);
  });
};

DevicesController.prototype.exportDeviceCsv = function(request, response) {
  var that = this;
  var userId = request.currToken.userID;
  var clientDataObj = {"userId" : userId}
  this.sDevice.getDevicesForCsv(clientDataObj , function(err, data) {
    response.attachment(global.__CONFIG__.calibration.exportCsvFileDownloadName);
    var options = {
      root: global.__CONFIG__.app_base_path + "../" + global.__CONFIG__.filesFolderName,
      dotfiles: 'deny',
      headers: {
        'x-timestamp': Date.now(),
        'x-sent': true
      }
    };
    response.sendFile(userId+global.__CONFIG__.calibration.allRecordsCsvFileName, options, function(err) {
      if (err) {
        that.sendResponse(new AppError(500,
          "There was a problem on the server.Please try again later", err), data, response);
      };

    });


  });
};

DevicesController.prototype.unExportedCsvRows = function(request, response) {
  var that = this;
  var userId = request.currToken.userID;
  response.attachment(global.__CONFIG__.calibration.unExportedLogFileName);
  var options = {
    root: global.__CONFIG__.app_base_path + "../" + global.__CONFIG__.filesFolderName,
    dotfiles: 'deny',
    headers: {
      'x-timestamp': Date.now(),
      'x-sent': true
    }
  };
  response.sendFile(userId+global.__CONFIG__.calibration.failureRecordsFileName, options, function(err) {
    if (err) {
      that.sendResponse(new AppError(500, "There was a problem on the server.Please try again later",
        err), data, response);
    };

  });

};
