emanateApp.service('CalibrationService', function() {
  var getDetsForCsv = function(cbAlways){
    
      $.fileDownload(__GLOBAL__.baseAPItoken +'export-devices/', {
              successCallback: function (url) { 
                err = {"hasError" : false,
                    "errorMessage" : "There was an error on the server"};
                cbAlways(err);
              },
              failCallback: function (responseHtml, url) { 
                err = {"hasError" : true,
                    "errorMessage" : "There was an error on the server"};
                cbAlways(err);
          
              },
              contentType : 'text/csv',
              httpMethod : 'GET'
          });
  };
  var getUnexportedRows = function(cbAlways){
    $.fileDownload(__GLOBAL__.baseAPItoken +'unexported-rows/', {
            successCallback: function (url) { 
              err = {"hasError" : false,
                  "errorMessage" : "There was an error on the server"};
              cbAlways(err);
            },
            failCallback: function (responseHtml, url) { 
              err = {"hasError" : true,
                  "errorMessage" : "There was an error on the server"};
              cbAlways(err);
        
            },
            contentType : 'text/csv',
            httpMethod : 'GET'
        });
  };
  
  var getCalibrationDetails = function(cbAlways){
      try {
          $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
            'url': __GLOBAL__.baseAPItoken + 'calibration/details',
            'type': 'GET'
          })).done(function(response) {
            if (response.status === 'success') {
              return cbAlways(response);
            } else {
              cbAlways();
            }
          }).fail(function(jqXHR, textStatus, errorThrown) {
            cbAlways(jqXHR, textStatus, errorThrown);
          }).always(function(res) {
            if (cbAlways) {
              cbAlways(res);
            }
          });
        } catch (e) {
          cbAlways(e, null, 500);
        }
  };
  
  return {
    getDetsForCsv : getDetsForCsv,
    getUnexportedRows : getUnexportedRows,
    getCalibrationDetails : getCalibrationDetails
  }
});
