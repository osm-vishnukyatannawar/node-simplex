emanateApp.service('OrganizationService', function() {
  var getOrganizationsByUserID = function(userID, cbProcess, cbAlways) {
    try {
      $.ajax($.extend(defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken + 'user/' + userID + '/organization/'
      })).done(function(response) {
        cbProcess(response);
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
  
  var getAllOrganizationsForUser = function(userID, cbProcess, cbAlways) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken+'user/'+userID+'/organizations'
      })).done(function(response) {
        if (response.status === 'success') {
          return cbProcess(response);
        } else {
          cbProcess();
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
  
  var getOrganizationByID = function(organizationID, cbProcess, cbAlways) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken+'organization/'+organizationID
      })).done(function(response) {
        if (response.status === 'success') {
          return cbProcess(response);
        } else {
          cbProcess();
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
  
  return {
    getOrganizationsByUserID: getOrganizationsByUserID,
    getAllOrganizationsForUser: getAllOrganizationsForUser,
    getOrganizationByID: getOrganizationByID
  };
});
