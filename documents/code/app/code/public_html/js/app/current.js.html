emanateApp.controller('CurrentController', function($scope, $document,
        CurrentService) {
  $document.ready(function() {
    CurrentService.getCurrentData(function(response, textStatus, errorThrown) {
      if (errorThrown) { return; }
      var source = {
        datatype: 'json',
        localdata: response,
        root: 'data',
        pagesize: 10
      };
      var dataAdapter = new $.jqx.dataAdapter(source);
      var $jqxCurrent = $('#16_dvCurrent');
      var gridConfig = $.extend({}, __GLOBAL__.defaultGridProperties, {
        source: dataAdapter,
        columns: [{
          text: 'Tag SN',
          dataField: 'tagSN',
          width: 90
        }, {
          text: 'Number of Samples',
          dataField: 'numSamples',
          width: 120
        }, {
          text: 'Current RMS',
          dataField: 'currentRms',
          width: 120
        }, {
          text: 'organizationID',
          dataField: 'organizationID',
          width: 120,
          hidden: true
        }, {
          text: 'Usage Info',
          dataField: 'powerPathUsageInfo',
          width: 120,
        }, {
          text: 'Start Timestamp',
          dataField: 'startTimestamp',
          width: 120,
        }, {
          text: 'Utilization',
          dataField: 'utilVal',
          width: 120,
        }]
      });

      gridConfig.columns = __GRID_HELPER__.processColumns(gridConfig.columns);
      __GRID_HELPER__.generateGridToolbars(gridConfig, $jqxCurrent);
      $jqxCurrent.jqxGrid(gridConfig);
    });
  });
});

emanateApp.service('CurrentService', function() {
  var getCurrentData = function(cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'current',
      'type': 'GET'
    }, cbProcess).done(function(response) {
      if (response.status === 'success') {
        cbProcess(response);
      } else {
        cbProcess();
      }
    }).fail(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  return {
    getCurrentData: getCurrentData
  };
});
