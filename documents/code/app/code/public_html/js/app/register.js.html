emanateApp.service('RegisterService', function() {
  var registerUser = function(user, cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPI + 'user/register/',
      'type': 'POST',
      'data': JSON.stringify(user)
    }, cbProcess).done(function(response) {
      cbProcess(response);
    }).error(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  var checkIfUserExists = function(emailID, cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPI + 'user/exists',
      'data': {
        emailID: emailID
      }
    }, cbProcess).done(function(response) {
      return cbProcess(true);
    }).error(function(jqXHR, textStatus, errorThrown) {
      if (jqXHR.status === 404) { return cbProcess(false); }
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  return {
    registerUser: registerUser,
    checkIfUserExists: checkIfUserExists
  };
});
