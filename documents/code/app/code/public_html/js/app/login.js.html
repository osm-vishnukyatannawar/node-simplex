var emanateApp = angular.module('emanateApp', []);

emanateApp.controller('LoginController', function($scope, LoginService) {
  $scope.login = {};
  $scope.isSubmitted = false;
  $scope.hasError = 0;
  $scope.isLoginPage = true;
  $scope.loginUser = function($login, isValid) {
    $scope.isSubmitted = true;
    $scope.hasError = 0;
    if (isValid) {
      LoginService.validateUser($login, cbLoginProcess, cbLoginComplete);
    }
  };

  function cbLoginProcess(response, textStatus, errorThrown) {
    if (errorThrown) {
      var respObj = getErrorResponseMsg(response);
      $scope.$apply(scopeApplyResponse(respObj, 1));
      return;
    }
    LoginService.saveUser(JSON.stringify(response.data));
    $scope.$apply(scopeApplyResponse(response, -1));
    if(response.data.isAdmin === '1') {
      window.location.replace('/dashboard-admin#/overview');
    } else {
      window.location.replace('/dashboard#/overview');
    }
    
  }

  function cbLoginComplete() {

  }

  function scopeApplyResponse(respObj, hasError) {
    $scope.msgToShow = respObj.message;
    $scope.hasError = hasError;
  }
});

emanateApp.service("LoginService", function() {
  var validateUser = function(user, cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPI + 'user/login/',
      'type': 'POST',
      'data': JSON.stringify(user)
    }, cbProcess).done(function(response) {
      if (response.status === 'success') {
        cbProcess(response);
      } else {
        cbProcess();
      }
    }).fail(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  var saveUser = function(data) {
    sessionStorage.setItem(__GLOBAL__.currUsrData, data);
  };

  return {
    validateUser: validateUser,
    saveUser: saveUser
  };
});
