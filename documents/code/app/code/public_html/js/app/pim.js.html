emanateApp.controller('PIMController', function($scope, $document, PIMService) {

  });
});

emanateApp.service("PIMService", function() {
  var getPIMData = function(cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'pim',
      'type': 'GET'
    }, cbProcess).done(function(response) {
      if (response.status === 'success') {
        cbProcess(response);
      } else {
        cbProcess();
      }
    }).fail(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  return {
    getPIMData: getPIMData
  };
});
