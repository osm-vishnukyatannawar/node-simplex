    $document.ready(function () {
        var $jqxDevices = $('#1_dvDevices');
        var source = {
                datatype: 'json',
                url: 'temp-url',
                sort: function () {
                  $jqxDevices.jqxGrid('updatebounddata', 'sort');
                   },
                  root: 'data',
                  beforeprocessing: function (data) {
                       source.totalrecords = data.numOfRecords;
                   },
                  pagesize : 10 
              };
        function getQueryParam(queryParams) {
          var request = {};
          request.limit = queryParams.pagesize;
          request.start = queryParams.recordstartindex;
          request.sortcol = queryParams.sortdatafield;
          request.sortby = queryParams.sortorder;
          request.startRecord = queryParams.pagenum * request.limit;
          request.endRecord = request.startRecord + request.limit;
          return request;
        }
        
        var dataAdapter = new $.jqx.dataAdapter(source , {
          loadServerData: function (serverdata, source, callback) {
            var requestData = getQueryParam(serverdata);
            var user = CurrentUserService.getUser();
            if (!user.currOrganization) {
                return;
            }
            DeviceService.getDevicesInfo(user.currOrganization,requestData,
            function (response) {
                dataAdapter.startIndex = requestData.startRecord;
                dataAdapter.endIndex = requestData.endRecord;
                if(source.beforeprocessing) {
                  source.beforeprocessing(response.data);
                }
                dataAdapter.loadjson(null, response.data, source);
                callback({
                    records: dataAdapter.records,
                    totalrecords: source.totalrecords
                });
            });
        }
    });
        
        
        var deviceGridCode = {

emanateApp.service("DeviceService", function() {
  var getDevicesInfo = function(orgId,queryParams,cbProcess, cbAlways) {
      try {
        $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
          'url': __GLOBAL__.baseAPItoken + 'organization/'+orgId+'/devices',
          'type': 'GET',
          'data': queryParams
        })).done(function(response) {
          if (response.status === 'success') {
            return  cbProcess(response);
          } else {
            cbProcess();
          }        
        }).fail(function(jqXHR, textStatus, errorThrown) {
          cbProcess(jqXHR, textStatus, errorThrown);
        }).always(function() {
          if (cbAlways) {
            cbAlways();
          }
        });
      } catch (e) {
        cbProcess(e, null, 500);
      }
    };
