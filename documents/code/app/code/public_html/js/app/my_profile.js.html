emanateApp.controller('MyProfileController', function($scope, $document, CurrentUserService,
    OrganizationService,UserService) {
  
  
  var $jqxOrganizations = null;
  var $jqxRecentLogins = null;
  var $jqxActiveSessions = null;
  
  var organizationsGridClass = {};
  var recentLoginsGridClass = {};
  var activeSessionsGridClass = {};
  
  var organizationsGridConfig = {
    columns: [{
          text: 'Organization name',
          dataField: 'organizationName',
          width: 200
        },{
          text: 'Organization address',
          dataField: 'organizationAddress',
          width: 230
        }, {
          text: 'Contact number',
          dataField: 'contactNumber',
          width: 100
        }, {
          text: 'Organization notes',
          dataField: 'organizationNotes',
          width: 230
        },{
          text: 'Organization Id',
          dataField: 'organizationID',
          width: 100,
          hidden: true
      }]  
  };
  
  var recentLoginsGridConfig = {
    columns: [{
          text: 'IP Address',
          dataField: 'IPAddress',
          width: 100
        }, {
          text: 'Location',
          dataField: 'Location',
          width: 200
        }, {
          text: 'Date/Time',
          dataField: 'DateTime',
          width: 230
       }]
  };
  
  var activeSessionsGridConfig = {
    columns: [{
            text: '#',
            dataField: 'Num',
            width: 30
          }, {
            text: 'Started',
            dataField: 'Started',
            width: 200
          }, {
            text: 'Location(IP Address)',
            dataField: 'Location',
            width: 200
          }, {
            text: 'Expires in',
            dataField: 'ExpiresIn',
            width: 100
          }, {
            text: 'Last active',
            dataField: 'LastActive',
            width: 130
          }, {
            text: 'User agent',
            dataField: 'UserAgent',
            width: 200
          }, {
            text: 'End session',
            dataField: 'EndSession',
            width: 100
      }]
  };
  
  $document.ready(function() {
    
    $jqxOrganizations = $('#1_dvOrganizations');
    $jqxRecentLogins = $('#1_dvRecentLogins');
    $jqxActiveSessions = $('#1_dvActiveSessions');
    
    organizationsGridClass = getInitializedGrid({
        config: organizationsGridConfig,
        element: $jqxOrganizations,
        ddlColContainer: '#organizationGridColContainer',
        ngScope: $scope
      });
    
    recentLoginsGridClass = getInitializedGrid({
        config: recentLoginsGridConfig,
        element: $jqxRecentLogins,
        ddlColContainer: '#recentLoginsGridColContainer',
        ngScope: $scope
      });
    
    activeSessionsGridClass = getInitializedGrid({
        config: activeSessionsGridConfig,
        element: $jqxActiveSessions,
        ddlColContainer: '#activeSessionsGridColContainer',
        ngScope: $scope
      });
    
    organizationsGridClass.initializeGrid();
    recentLoginsGridClass.initializeGrid();
    activeSessionsGridClass.initializeGrid();
    
    
    $scope.curUser = CurrentUserService.getUser();
    $scope.curUserId = $scope.curUser.userID;
    OrganizationService.getAllOrganizationsForUser($scope.curUserId,
        organizationsGridClass.refreshGridData);
    UserService.getRecentLoginsForUser(recentLoginsGridClass.refreshGridData);
    UserService.getActiveSessionsForUser(activeSessionsGridClass.refreshGridData);
    
    $(".show-hide-form").click(function() {
      var $changeEmailForm = $('.email-form');
      if ($changeEmailForm.hasClass('hidden')) {
        $(this).text('Hide');
        $changeEmailForm.removeClass('hidden');
        $changeEmailForm.addClass('show');
      } else if ($changeEmailForm.hasClass('show')) {
        $(this).text('View');
        $changeEmailForm.removeClass('show');
        $changeEmailForm.addClass('hidden');
      }
    });
    
    
  });
});
