function clearSessionStorage() {
  var i = sessionStorage.length;
  while (i--) {
    var key = sessionStorage.key(i);
    sessionStorage.removeItem(key);
  }
}

function logOutUser() {
  clearSessionStorage();
  window.location.replace('/');
}
