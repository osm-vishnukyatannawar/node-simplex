emanateApp.service('UserService', function() {
  var getUsersByOrganization = function(organizationID, queryParams, callback) {
    $.ajax(
      $.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken + 'organization/' + organizationID + '/users',
        'data': queryParams
      })).done(function(response) {
      if (response.status === 'success') {
        callback(response.data);
      }
    });
  };

  var saveUser = function(userObj, callback) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'user/add',
      'type': 'POST',
      'data': JSON.stringify(userObj)
    }, callback).done(function(response) {
      callback(response.data);
    });
  };

  return {
    getUsersByOrganization: getUsersByOrganization,
    saveUser: saveUser
  };
});
