    RoleService.getRoles(function(response) {
      var roles = {};
      for (var i = 0, len = response.length; i < len; ++i) {
        roles[response[i].roleID] = response[i].roleName;
      }
      $timeout(function() {
        $scope.roles = roles;
      });
    });

    userGridClass = getInitializedGrid({
      config: userGridConfig,
      element: $jqxUsers,
      ddlColContainer: '#userGridColContainer',
      ngScope: $scope
    });

    $scope.hasError = 0;
    $scope.roles = {};
    $scope.currUser = CurrentUserService.getUser();
    $scope.deployments = $scope.currUser.deployments;
    $scope.user = {};
    $scope.user.isSubmitted = false;
    $scope.user['roles'] = [];
    $scope.addRoleRow = function() {
      $scope.user.roles.push({
        deploymentID: '',
        roleID: ''
      });
    };

    $scope.removeRole = function(index) {
      if ($scope.user.roles.length > index) {
        $scope.user.roles.splice(index, 1);
      }
    };

    var $userSaveBtn = $('#1_btnSaveUser')[0];
    $scope.saveUser = function(user, isValidForm) {
      $scope.user.isSubmitted = true;
      $scope.msgToShow = '';
      $scope.hasError = 0;
      var l = Ladda.create($userSaveBtn);
      if (!isValidForm) {
        return;
      }
      var roles = $scope.user['roles'];
      if (roles.length === 0) {
        $scope.hasError = 1;
        $scope.msgToShow = __APP_MESSAGE__.userDepRoleErr;
        return;
      }

      var uniqueDeps = [];
      for (var i = 0; i < roles.length; ++i) {
        if (uniqueDeps.indexOf(roles[i].deploymentID) === -1) {
          uniqueDeps.push(roles[i].deploymentID);
        } else {
          $scope.hasError = 1;
          $scope.msgToShow = __APP_MESSAGE__.usrDuplicateDeployment;
          return;
        }
      }
      l.start();
      user.organizationID = $scope.currUser.currOrganization;
      UserService.saveUser(user, function(response, textStatus,
        errorThrown) {
        l.stop();
        if (errorThrown) {
          var respObj = getErrorResponseMsg(response);
          $scope.$apply(scopeApplyResponse(respObj, 1));
          return;
        }
        $timeout(function() {
          $scope.clearUser();
        });
        $('#1_userDialog').modal('toggle');
        UserService.getUsersByOrganization($scope.currUser.currOrganization, {},
          userGridClass.refreshGridData);
      });
    };

    $scope.clearUser = function() {
      $scope.user['roles'] = [];
      $scope.user.emailID = '';
      $scope.user.username = '';
    };

    userGridClass.initializeGrid();
    UserService.getUsersByOrganization($scope.currUser.currOrganization, {},
      userGridClass.refreshGridData);
  });

  function scopeApplyResponse(respObj, hasError) {
    $scope.msgToShow = respObj.message;
    $scope.hasError = hasError;
  }
});

emanateApp.service('UserService', function() {
  var getUsersByOrganization = function(organizationID, queryParams,
    callback) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'organization/' +
        organizationID + '/users',
      'data': queryParams
    }, callback).done(function(response) {
      if (response.status === 'success') {
        callback(response);
      }
    });
  };

  var getAllUsers = function(callback) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'users/',
      'type': 'GET'
    }, callback).done(function(response) {
      callback(response.data);
    });
  };

  var saveUser = function(userObj, callback) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'user/add',
      'type': 'POST',
      'data': JSON.stringify(userObj)
    }, callback).done(function(response) {
      callback(response.data);
    }).fail(function(jqXHR, textStatus, errorThrown) {
      callback(jqXHR, textStatus, errorThrown);
    });
  };
