var emanateApp = angular.module('emanateApp', []);

emanateApp.controller('ForgotPassController', function($scope,
        ForgotPassService) {
  $scope.forgotPassFormVal = {};
  $scope.isSubmitted = false;
  $scope.hasError = 0;
  $scope.forgotPass = function($forgotPassFormVal, isValid) {
    $scope.hasError = 0;
    $scope.isSubmitted = true;
    if (isValid) {
      ForgotPassService.validateEmail($forgotPassFormVal, cbForgotPassProcess, cbForgotPassComplete);
    }
  };

  function cbForgotPassProcess(response, textStatus, errorThrown) {
    if (errorThrown) {
      var respObj = getErrorResponseMsg(response);
      $scope.$apply(scopeApplyResponse(respObj, 1));
      return;
    }
    $scope.$apply(scopeApplyResponse(response, -1));
    window.location.replace(__GLOBAL__.url.dashboard);
  }

  function cbForgotPassComplete() {

  }

  function scopeApplyResponse(respObj, hasError) {
    $scope.msgToShow = respObj.message;
    $scope.hasError = hasError;
  }
});

emanateApp.service("ForgotPassService", function() {
  var validateEmail = function(email, cbProcess, cbAlways) {
    try {
      makeAjaxCall({
        'url': __GLOBAL__.baseAPI + 'user/password/',
        'type': 'POST',
        'data': JSON.stringify(email)
      }, cbProcess).done(function(response) {
        if (response.status === 'success') {
          cbProcess(response);
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
  var saveUser = function(data) {
    sessionStorage.setItem(__GLOBAL__.currUsrData, data);
  };

  return {
    validateEmail: validateEmail,
    saveUser: saveUser
  };
});
