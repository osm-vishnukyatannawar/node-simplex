emanateApp.service("HistogramService", function() {
  var getHistogramData = function(cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'histogram',
      'type': 'GET'
    }, cbProcess).done(function(response) {
      if (response.status === 'success') {
        cbProcess(response);
      } else {
        cbProcess();
      }
    }).fail(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  return {
    getHistogramData: getHistogramData
  };
});
