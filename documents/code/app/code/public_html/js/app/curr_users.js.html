emanateApp.service('CurrentUserService', function() {
  var user = {};

  var getUser = function() {
    return user;
  };

  var loadUser = function() {
    try {
      user = JSON.parse(sessionStorage.getItem(__GLOBAL__.currUsrData));
      if (!user) { throw new Error('Not logged in'); }
    } catch (e) {
      return false;
    }
  };

  var getUserFromServer = function(userID, cbProcess, cbAlways) {
    makeAjaxCall({
      'url': __GLOBAL__.baseAPItoken + 'user/:id'
    }).done(function(response) {
      cbProcess(response);
    }).fail(function(jqXHR, textStatus, errorThrown) {
      cbProcess(jqXHR, textStatus, errorThrown);
    }).always(function() {
      if (cbAlways) {
        cbAlways();
      }
    });
  };

  var refresh = function() {
    getUserFromServer(user.userID, function(response, textStatus, errorThrown) {
      sessionStorage.setItem(__GLOBAL__.currUsrData, JSON
              .stringify(response.data));
    });
  };

  var clearUser = function() {
    var i = sessionStorage.length;
    while (i--) {
      var key = sessionStorage.key(i);
      sessionStorage.removeItem(key);
    }
  };

  var logoutUser = function(cbProcess, cbAlways) {
    try {
      makeAjaxCall({
        'type': 'POST',
        'data': JSON.stringify({
          'userID': user.userID
        }),
        'url': __GLOBAL__.baseAPItoken + 'user/logout'
      }).done(function(response) {
        cbProcess(response);
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };

  var updateUser = function(user) {

  };
  
  var isAdmin = function() {
    var user = this.getUser();
    if(user.isAdmin === '1') {
      return true;
    }
    return false;
  };
  
  return {
    loadUser: loadUser,
    isAdmin : isAdmin,
    clearUser: clearUser,
    getUser: getUser,
    logoutUser: logoutUser,
    getRemoteUser: getUserFromServer,
    refresh: refresh
  };
});
