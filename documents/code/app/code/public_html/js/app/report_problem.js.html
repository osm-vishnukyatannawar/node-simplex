emanateApp.controller('ReportProblemController', function($scope, $document, $upload, $timeout, $filter,
  CurrentUserService, OrganizationService, DeploymentService) {
  $document.ready(function() {
    $scope.hasError = 0;
    $scope.isSubmitted = false;
    $scope.reportProblemFormVal = {};
    $scope.fileNames =[];

    $scope.reportProblem = function($reportProblemFormVal, isValid) {
      $scope.isSubmitted = true;
      $scope.hasError = 0;
      var $reportBtn = $('#43_btnReportProblem')[0];
      var loadingInd = Ladda.create($reportBtn);
      loadingInd.start();
      if (!isValid) {
        loadingInd.stop();
        return;
      } else {

        var files = $reportProblemFormVal.reportFile;
        if(!angular.isUndefined(files) && files !== null) {
          for(var i = 0; i < files.length; ++i) {
            $scope.fileNames.push(files[i].name.split('.')[0]);
          }
        }
        $scope.upload = $upload.upload({
          url: __GLOBAL__.baseAPItoken + 'problem/',
          method: 'POST',
          data: $reportProblemFormVal,
          file: files,
          fileFormDataName: $scope.fileNames
        });

        $scope.upload.then(function(response) {
          $timeout(function() {
            if (response.status === 200) {
              $scope.clearReportDialog($scope.reportProblemFormVal);
              loadingInd.stop();
              $('#48_reportProblemDialog').modal('toggle');
            } else {
              loadingInd.stop();
              $scope.hasError = 1;
              $scope.msgToShow = "There was an error on the server please try again later.";
            }

          });
        }, function(response) {
          if (response.status > 0) {
          loadingInd.stop();
          $scope.hasError = 1;
          $scope.msgToShow = response.data.message;
          }
        });
      }
    };

    $scope.clearReportDialog = function($reportProblemFormVal) {
      $reportProblemFormVal.description = '';
      $reportProblemFormVal.reportFile = null;
    };

    $scope.populateValues = function($reportProblemFormVal) {
      $scope.currUserData = CurrentUserService.getUser();
      $reportProblemFormVal.userID = $scope.currUserData.userID;
      $reportProblemFormVal.organizationID = $scope.currUserData.currOrganization;
      $reportProblemFormVal.deploymentID = $scope.currUserData.currDeployment;
      $scope.populateUsername($reportProblemFormVal);
      if ($scope.currUserData.currOrganization) {
        OrganizationService.getOrganizationByID($scope.currUserData.currOrganization,
          $scope.populateOrganization);
      } else {
        $reportProblemFormVal.organization = '';
      }
      if ($scope.currUserData.currDeployment) {
        DeploymentService.getCurrentDeployment($scope.currUserData.currDeployment,
          $scope.populateDeployment);
      } else {
        $reportProblemFormVal.currDeployment = '';
      }
      $scope.populateCurrDate($reportProblemFormVal);
    };

    $scope.populateUsername = function($reportProblemFormVal) {
      $reportProblemFormVal.username = $scope.currUserData.username;
    };

    $scope.populateOrganization = function(response, textStatus, errorThrown) {
      if (response.status === 'success') {
        $scope.$apply(function() {
          $scope.reportProblemFormVal.organization = response.data.organizationName;
        });
      }
    };

    $scope.populateDeployment = function(response, textStatus, errorThrown) {
      if (response.status === 'success') {
        $scope.$apply(function() {
          $scope.reportProblemFormVal.deployment = response.data.deploymentName;
        });
      }
    };

    $scope.populateCurrDate = function($reportProblemFormVal) {
      var date = new Date();
      $reportProblemFormVal.dateAndTime = date;
    };

    $('#sbLnkReportAProblem').click(function(e) {
      e.preventDefault();
      $('#48_reportProblemDialog').modal();
    });

    $('#48_reportProblemDialog').on('shown.bs.modal', function() {
      $scope.$apply(function() {
        $scope.populateValues($scope.reportProblemFormVal);
      });
    });
  });
});
