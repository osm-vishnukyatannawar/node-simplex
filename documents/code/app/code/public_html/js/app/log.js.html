emanateApp.controller('LogController', function($scope, $document, LogService) {
  
  var $jqxLogs = null;
  
  var logsGridClass = {};
  
  var logsGridConfig = {
    columns: [{
          text: 'Username',
          dataField: 'username',
          width: 200
        },{
          text: 'Log type',
          dataField: 'logType',
          width: 200
        },{
          text: 'Log Description',
          dataField: 'logDescription',
          width: 300
        }, {
          text: 'Created on',
          dataField: 'createdOn',
          width: 200
        },{
          text: 'Log Id',
          dataField: 'LogID',
          width: 100,
          hidden: true
      }]  
  };
  
  $document.ready(function() {
    
    $jqxLogs = $('#43_dvLog');
    
    logsGridClass = getInitializedGrid({
        config: logsGridConfig,
        element: $jqxLogs,
        ddlColContainer: '#logGridColContainer',
        ngScope: $scope
      });
    
    logsGridClass.initializeGrid();
    
    LogService.getLogTypes(bindLogs);
    
    function bindLogs(response, textStatus, errorThrown) {
      if(!errorThrown) {
        $scope.$apply(function(){
          $scope.logs = response.data;
        });
      }
    }
    
    $scope.logType = '1';
    LogService.getLogsBasedOnType($scope.logType, logsGridClass.refreshGridData);
    
    $scope.filterLog = function() {
      LogService.getLogsBasedOnType($scope.logType, logsGridClass.refreshGridData);
    };
    
  });
});

emanateApp.service('LogService', function() {
  
  var getLogsBasedOnType = function(logType, cbProcess, cbAlways) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken+'log/'+ logType
      })).done(function(response) {
        if (response.status === 'success') {
          return cbProcess(response);
        } else {
          cbProcess();
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      }).always(function() {
        if (cbAlways) {
          cbAlways();
        }
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
  
  var getLogTypes = function(cbProcess) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken+'logTypes/'
      })).done(function(response) {
        if (response.status === 'success') {
          return cbProcess(response);
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbProcess(jqXHR, textStatus, errorThrown);
      });
    } catch (e) {
      cbProcess(e, null, 500);
    }
  };
    
  return {
    getLogsBasedOnType: getLogsBasedOnType,
    getLogTypes: getLogTypes
  }
  
});
