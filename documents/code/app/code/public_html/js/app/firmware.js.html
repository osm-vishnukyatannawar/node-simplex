emanateApp.service("FirmwareService", function() {
  var getFirwareInfo = function(cbAlways) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken + 'firmware/',
        'type': 'GET'
      })).done(function(response) {
        if (response.status === 'success') {
          return cbAlways(response);
        } else {
          cbAlways();
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbAlways(jqXHR, textStatus, errorThrown);
      }).always(function(res) {
        if (cbAlways) {
          cbAlways(res);
        }
      });
    } catch (e) {
      cbAlways(e, null, 500);
    }
  };
  
  var getFirwareVersions = function(cbAlways) {
    try {
      $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
        'url': __GLOBAL__.baseAPItoken + 'firmware/versions',
        'type': 'GET'
      })).done(function(response) {
        if (response.status === 'success') {
          return cbAlways(response);
        } else {
          cbAlways();
        }
      }).fail(function(jqXHR, textStatus, errorThrown) {
        cbAlways(jqXHR, textStatus, errorThrown);
      }).always(function(res) {
        if (cbAlways) {
          cbAlways(res);
        }
      });
    } catch (e) {
      cbAlways(e, null, 500);
    }
  };

  return {
    getFirwareInfo: getFirwareInfo,
    getFirwareVersions: getFirwareVersions
  };
});

  return {
    require: 'ngModel',
    link: function(scope, el, attrs, ngModel) {
      el.bind('change', function() {
        scope.$apply(function() {
          ngModel.$setViewValue(el.val());

        });

    }
  };
