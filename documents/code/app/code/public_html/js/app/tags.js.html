    emanateApp.service("TagService", function() {
        var getTagsInfo = function(orgId,queryParams,cbProcess, cbAlways) {
            try {
              $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
                'url': __GLOBAL__.baseAPItoken + 'organization/'+orgId+'/tags',
                'type': 'GET',
                'data': queryParams
              })).done(function(response) {
                if (response.status === 'success') {
                  return  cbProcess(response);
                } else {
                  cbProcess();
                }        
              }).fail(function(jqXHR, textStatus, errorThrown) {
                cbProcess(jqXHR, textStatus, errorThrown);
              }).always(function() {
                if (cbAlways) {
                  cbAlways();
                }
              });
            } catch (e) {
              cbProcess(e, null, 500);
            }
          };
          
          var getTagInfo = function(tagID, cbProcess, cbAlways) {
              try {
                $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
                  'url': __GLOBAL__.baseAPItoken + 'tag/'+tagID,
                  'type': 'GET'           
                })).done(function(response) {
                  if (response.status === 'success') {
                    cbProcess(response);
                  } else {
                    cbProcess();
                  }        
                }).fail(function(jqXHR, textStatus, errorThrown) {
                  cbProcess(jqXHR, textStatus, errorThrown);
                }).always(function() {
                  if (cbAlways) {
                    cbAlways();
                  }
                });
              } catch (e) {
                cbProcess(e, null, 500);
              }
            };  
          
          return {      
            getTagsInfo: getTagsInfo,
            getTagInfo : getTagInfo
            };    
          
      
    });
