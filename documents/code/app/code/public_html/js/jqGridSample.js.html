$( 'document' ).ready(function(){
  var $jqxUsers = $('#1_dvUsers');
  var source = {
    datatype: 'json',
    url: 'temp-url',
    filter: function() {},
    sort: function() {
      $jqxUsers.jqxGrid('updatebounddata', 'sort');
    },
    root: 'data',
    beforeprocessing: function(data) {
      source.totalrecords = data.numOfRecords;
    },
    pagesize: 10
  };

  function getQueryParam(queryParams) {
    var request = {};
    request.limit = queryParams.pagesize;
    request.start = queryParams.recordstartindex;
    request.sortcol = queryParams.sortdatafield;
    request.sortby = queryParams.sortorder;
    request.startRecord = queryParams.pagenum * request.limit;
    request.endRecord = request.startRecord + request.limit;
    return request;
  }
  var orgID = "cd28276d-668b-11e4-945d-000c29609978"
  var dataAdapter = new $.jqx.dataAdapter(source, {
    loadServerData: function(serverdata, source, callback) {
      var requestData = getQueryParam(serverdata);
      getUsersByOrganization(orgID, requestData,

        function(response, textStatus, errorThrown) {
          dataAdapter.startIndex = requestData.startRecord;
          dataAdapter.endIndex = requestData.endRecord;
          if (source.beforeprocessing) {
            source.beforeprocessing(response);
          }
          dataAdapter.loadjson(null, response, source);
          callback({
            records: dataAdapter.records,
            totalrecords: source.totalrecords
          });
        });
    }
  });
 function getUsersByOrganization(organizationID, queryParams, callback) {
    var baseAPItoken = 'cc2ca458c67041dc'
    $.ajax($.extend(__GLOBAL__.defaultAjaxParams, {
      'url': '/api/v1/'+baseAPItoken+'/organization/'+organizationID+'/users',
      'data': queryParams
    })).done(function(response) {
      if (response.status === 'success') {
        callback(response.data);
      }
    });
  };
  var userGridCode = {
    renderGridRows: function(objToBind) {
      objToBind.startindex = dataAdapter.startIndex;
      objToBind.endindex = dataAdapter.endIndex;
      return objToBind.data;
    },
    renderToolbars: function(toolbar) {
      var container = $('<div style="margin: 5px;"></div>');
      var colButton = $('<button id="1_btnAddCols" class="btn btn-xs pull-right"><span class="glyphicon glyphicon-plus"></span></button>');
      container.append(colButton);
      toolbar.append(container);
    }
  };

  var gridConfig = $.extend({}, __GLOBAL__.defaultGridProperties, {
    source: dataAdapter,
    rendertoolbar: userGridCode.renderToolbars,
    rendergridrows: userGridCode.renderGridRows,
    columns: [{
      text: 'ID',
      dataField: 'userID',
      width: 120
    }, {
      text: 'Username',
      dataField: 'username',
      width: 200
    }, {
      text: 'Contact Number',
      dataField: 'contactNumber',
      width: 150
    }, {
      text: 'Email ID',
      dataField: 'emailID',
      width: 220
    }, {
      text: 'Deployments',
      dataField: 'deployments',
      width: 240
    }, {
      text: 'Timezone',
      dataField: 'userTimeZone',
      width: 180
    }, {
      text: 'Created On',
      dataField: 'createdOn',
      width: 150
    }, {
      text: 'Modified On',
      dataField: 'modifiedOn',
      width: 150
    }]
  });

  $jqxUsers.jqxGrid(gridConfig).on('bindingcomplete', function(event) {
    $jqxUsers.jqxGrid('hidecolumn', 'userID');
    $jqxUsers.jqxGrid('hidecolumn', 'deployments');
    $jqxUsers.jqxGrid('hidecolumn', 'modifiedOn');
    $jqxUsers.jqxGrid('hidecolumn', 'createdOn');
    $jqxUsers.jqxGrid('clearselection');
  });

  
});
