var __GRID_HELPER__ = {
  columnsrenderer: function(value) {
    return '<div style="margin-top:5px">' + value + '</div>';
  },
  processColumns: function(gridColumns) {
    for (var i = 0, len = gridColumns.length; i < len; ++i) {
      if (!gridColumns[i].hasOwnProperty('renderer')) {
        gridColumns[i].renderer = __GRID_HELPER__.columnsrenderer;
      }
      if (!gridColumns[i].hasOwnProperty('menu')) {
        gridColumns[i].menu = false;
      }      
    }
    return gridColumns;
  },
  generateGridToolbars : function(gridConfig, $gridObj) {
    var columns = gridConfig.columns;    
    gridConfig.rendertoolbar = function(toolbar) {
      var perColumnContent = 8;
      var columnCount = 1;      
      if(columns.length > perColumnContent) {
        columnCount = 2;        
      }
      var container = $('<div style="margin: 5px;" class="dropdown clearfix pull-right"></div>');
      var colButton = $('<button id="1_btnAddCols" type="button" data-toggle="dropdown" class="btn btn-xs"><span class="glyphicon glyphicon-plus"></span></button>');      
      var currColumn = 0;
      var finalArray = [];
      for(var i = 0, len = columnCount; i != len; ++i) {
        finalArray[i] = ['<ul class="dropdown-menu dropdown-menu-form">'];
      }
      for(var i = 0, len = columns.length; i != len; ++i) {
        currColumn = i % columnCount;
        finalArray[currColumn].push('<li><label><input type="checkbox"'
                + ' data-colname="' + columns[i].dataField + '"/>' 
                + columns[i].text + '</label></li>');        
      }
      var finalHTMLArr = [];
      for(var i = 0, len = columnCount; i != len; ++i) {
        finalArray[i].push('</ul>');
        finalHTMLArr[i] = [finalArray[i].join("")];
      }
      var finalHtml = finalHTMLArr.join("");
      container.append(colButton).append(finalHtml);
      toolbar.append(container);
      __GRID_HELPER__.addColChkHandler($gridObj);
    };
  },
  addColChkHandler : function($gridObj) {    
    $gridObj.find('.dropdown-menu-form').click(function(e){
      e.stopPropagation();
    });
    $gridObj.find('.dropdown-menu-form input[type="checkbox"]').click(function(e) {
      
    });
  },
  getQueryParam : function (queryParams) {
    var request = {};
    request.limit = queryParams.pagesize;
    request.start = queryParams.recordstartindex;
    request.sortcol = queryParams.sortdatafield;
    request.sortby = queryParams.sortorder;
    request.startRecord = queryParams.pagenum * request.limit;
    request.endRecord = request.startRecord + request.limit;
    return request;
  }  
};

function getErrorResponseMsg(responseObj) {
  var msg = __APP_MESSAGE__.unknownErr;
  var errObj = {};
  if (responseObj.hasOwnProperty('responseJSON')) {
    var status = responseObj.responseJSON.status;
    if (status === "fail") {
      msg = responseObj.responseJSON.message;
      errObj = responseObj.responseJSON.data;

    } else if (status === 'error') {
      msg = __APP_MESSAGE__.serverErr;
    }
  }
  return {
    'message': msg,
    'error': errObj
  };
}

function makeAjaxCall(ajaxObj, cbProcess) {
  try {
    return $.ajax($.extend(__GLOBAL__.getDefaultAjaxParams(), ajaxObj));
  } catch (e) {
    cbProcess(e, null, 500);
  }
