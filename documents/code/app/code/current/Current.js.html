CurrentData.prototype.insertCurrentData = function(currentData, cb) {
  var transactionID = null;
  if (currentData.hasOwnProperty('transactionID')) {
    transactionID = currentData.transactionID;
    currentData = currentData.data;
  }
  var sqlQuery =  'INSERT INTO tag_current_data (`tag_current_data_recid`, `cassandra_dump_rec_id`, `tag_srno`, `organization_id`,'
      + '`start_timestamp`,`num_samples`, `created_on`, `created_by`)'
      + ' VALUES (:id, :dmpRecID, :tagSN, :organizationID, :startTimestamp, :numSamples, :createdOn, :createdBy);';  
  
  this.query({
    query: sqlQuery,
    data : currentData,
    transactionID : transactionID,
    cb: function(err, data) {
      if(err) {
        return cb(err,{});
      }
      return cb(null, {
        currentDataID : currentData.id
      });
    }  
  });
};

CurrentData.prototype.insertCurrentDataUtil = function(currentDataUtil, cb) {
  var transactionID = null;
  if (currentDataUtil.hasOwnProperty('transactionID')) {
    transactionID = currentDataUtil.transactionID;
    currentDataUtil = currentDataUtil.data;   // Array of structures
  }
  var queryObj = this.getCurrentDataUtilMappingQuery(currentDataUtil);  
  this.query({
    query : queryObj.query,
    data : queryObj.data,
    transactionID : transactionID,
    cb: function(err, data) {
        if (err) { return cb(err,{}); }
        if(currentDataUtil.length === data.affectedRows) {
          return cb(null, {
            success : true          
          });
        }
        return cb(null, {
          success : false
        });
      }
  });
};

CurrentData.prototype.getCurrentDataUtilMappingQuery = function(currentDataUtil) {
    var sqlQuery = 'INSERT INTO tag_current_util_data ( `tag_current_util_data_recid`,'
      + '`current_rms`, `util_val`, `usage_state_id_enum`, `tag_current_data_recid`,`created_on`,`created_by`)'
      + ' VALUES ';
    var currDate = this.dbUtils.toMySQLDateTime();
    var finalData = {
        currentDataID: currentDataUtil.currentDataID,
      createdOn: currDate,
      createdBy: 'NodeJS'
    };
    var propNames = ['currentRms', 'utilVal','usageState'];
    var queryObj = this.getMultipleInsertQuery({
      initialQuery: sqlQuery,
      staticData: finalData,
      propNames: propNames,
      staticCols: [':currentDataID', ':createdOn',':createdBy'],
      data: currentDataUtil,
      hasPrimary: true
    });
    return queryObj; 
};

CurrentData.prototype.getCurrentData = function(cb) {
  var sqlQuery = 'SELECT header.tag_srno AS tagSN, header.organization_id AS organizationID, header.start_timestamp AS startTimestamp, ' 
    + 'header.num_samples AS numSamples, detail.current_rms AS currentRms, detail.util_val AS utilVal, '
    + '(SELECT Lookup_value_vc FROM tag_lookup_table WHERE Lookup_Key = \'powerPathUsageInfo\' '
    + 'AND lookup_value_id = detail.usage_state_id_enum) AS powerPathUsageInfo '
    + 'FROM tag_current_data header, tag_current_util_data detail '
    + 'WHERE header.tag_current_data_recid = detail.tag_current_data_recid;';
  this.getResults({
    query :sqlQuery,
    data : {},
    cb : cb
  });
};
