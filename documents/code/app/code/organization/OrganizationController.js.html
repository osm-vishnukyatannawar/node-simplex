var User = require(__CONFIG__.app_code_path + 'users/User');

var logger = require(__CONFIG__.app_base_path + 'logger');
var Controller = require(__CONFIG__.app_base_path + 'lib/controller');
var util = require("util");
var Tags = require(__CONFIG__.app_code_path + 'tags/Tag');
var Organization = require(__CONFIG__.app_code_path + 'organization/Organization');

function OrganizationController(app) {
  Controller.call(this);
  app.httpGet('organization/:id/deployments',
    this.getDeployments.bind(this));
  app.httpGet('organization/:id/devices',
    this.getDevices.bind(this));
  app.httpGet('organization/:id/tags',
    this.getTags.bind(this));
  app.httpGet('user/:id/organizations',
    this.getUserOrganizations.bind(this));
  app.httpGet('organization/:id',
  this.getOrganizationByID.bind(this));
}

util.inherits(OrganizationController, Controller);

OrganizationController.prototype.getDeployments = function(request, response) {
  var that = this;
  var id = request.params.id;
  this.Deployment = new Deployment();
  this.Deployment.getByOrganizationID(id, function(err, data) {
    that.sendResponse(err, data, response);
  });
};
