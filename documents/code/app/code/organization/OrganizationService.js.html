OrganizationService.prototype.getUserOrganizations = function(userID, cntrlCb) {
  var mOrganization = new Organization();
  mOrganization.getAllDetailsByUserID(userID, function(err, data) {
    if (err) {
      return cntrlCb(err, {});
    }
    return cntrlCb(err, OrganizationServiceHelper.formatOrganizationDetails(data));
  });
};
