¶ Get organization short name

Returns the organization short name. To create short name, the organization name is trimmed, spaces in between are replaced with _ and first four characters of the organization name (converted to upper case) are taken. In case the organization name is shorter than four characters, underscores are appended at the end.

Params
organizationName {string} The organization name for which the short name must be returned
cb {object} Callback method 

Organization.prototype.getShortOrganizationName = function(organizationName, cb) {
  var orgName = organizationName.trim().replace(/ /g,'_') + '____';
  orgName = orgName.substr(0,4).toUpperCase();
  return orgName;
};

Organization.prototype.checkIfOrgExists = function(orgID, cb) {
  if (__.isEmpty(orgID)) { return cb(new AppError(this
          .getStatusCode('badRequest'), 'Please send a valid organization ID',
          {})); }
  var sqlQuery = 'SELECT organization_recID AS organizationID FROM organization_m WHERE organization_id = ?';
  this.getResult({
    query : sqlQuery,
    data : [orgID],
    cb : function(err, data) {
      if(err) {
        return cb(err);
      } 
      if(__.isEmpty(data)) {
        return cb(null, false);
      } else {
        return cb(null, data.organizationID);
      }
    }
  });
};
