Organization.prototype.getByUserID = function(userID, cb) {
  if (__.isEmpty(userID)) { return cb(new AppError(this
          .getStatusCode('badRequest'), 'Please send the user ID.', {
    'userID': 'Please send the user ID.'
  })); }
  var sqlQuery = 'SELECT DISTINCT \
      om.organization_recid AS organizationRecID, \
      om.organization_name AS organizationName \
    FROM user_role_det AS urd \
      INNER JOIN user_m AS um \
        ON um.user_recid = urd.user_recid \
      INNER JOIN deployment_m AS dm \
        ON dm.deployment_recid = urd.deployment_recid \
      INNER JOIN organization_m AS om \
        ON dm.organization_recid = om.organization_recid \
    WHERE urd.is_active = 1 \
        AND um.is_active = 1 \
        AND dm.is_active = 1 \
        AND om.is_Active = 1 \
        AND um.user_recid = ? \
    ORDER BY om.created_on;';

  this.getResults({
    query: sqlQuery,
    data: [userID],
    cb: function(err, data) {
      cb(err, data);
    }
  });
};

  var sqlQuery = 'SELECT `organization_recid` AS organizationID, \
    `organization_id` AS ID, \

    `license_recid` AS licenseRecID, \
    `country_recid` AS countryID, \
    `contact_number` AS contactNumber, \
    `address` AS address, \
    `organization_url` AS organizationURL, \
    `notes` AS notes, \
    `is_Active` AS isActive, \
    `created_on` AS createdOn, \
