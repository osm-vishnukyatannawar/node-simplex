¶ Format organization details

Formats the organization object, groups the country and license details.

Params
data {object} organization object to be formatted returns formatted organization object 
  var formatOrganizationDetails = function(data) {
    var singleElement = {};
    var formattedData = [];
    var deleteElements = ['countryRecId','countryName','licenseRecId','licenseName','maxTags','maxUsers'];
    
     while (data.length > 0) {
        
        singleElement = data.shift();
        singleElement.country = {
          "id" : singleElement.countryRecId,
          "name": singleElement.countryName
        };
        singleElement.license = {
              "id" : singleElement.licenseRecId,
              "name": singleElement.licenseName,
              "maxTags" : singleElement.maxTags,
              "maxUsers" : singleElement.maxUsers
          };
        singleElement.organizationURL = __CONFIG__.app_base_url + singleElement.organizationURL;
        
        for (var i = 0; i < deleteElements.length; i++) { 
          delete singleElement[deleteElements[i]];
        }
   
        formattedData.push(singleElement);
     }
     return formattedData;
  };
  
  return {
      formatOrganizationDetails : formatOrganizationDetails   
  };
};
