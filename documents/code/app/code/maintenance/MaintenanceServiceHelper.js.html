var S = require('string'); 
var __ = require('underscore');

var MaintenanceServiceHelper = function() {
  var normalizeTagData = function(tagData) {
    var responseData = __.defaults({}, tagData);
    if (tagData.hasOwnProperty('SN')) { 
      responseData.tagSN = tagData.SN;
    }
    if(tagData.hasOwnProperty('orgID')) {
      responseData.organizationID = tagData.orgID;
    }
    responseData.data = tagData.data;
    responseData.type = tagData.type;
    return responseData;
  };
  
  var formatPendingEvents = function(pendingEvents) {
    var formattedEvents = {};
    
    for (var i=0;i < pendingEvents.length; i++) {
      formattedEvents[pendingEvents[i].eventTypeID] = pendingEvents[i].data;
    }
    return formattedEvents;
  };
  
  var isDefaultMaintObj = function(serviceObj) {
    if(serviceObj.tagSN === __CONFIG__.maintenance.default_value_tag_sn) {
      return true;
    }
    if(serviceObj.organizationID === __CONFIG__.maintenance.default_value_org) {
      return true;
    }
    if(serviceObj.type === __CONFIG__.maintenance.default_value_type) {
      return true;
    }
    return false;
  };
    
  return {
    normalizeTagData: normalizeTagData,
    formatPendingEvents: formatPendingEvents,
    isDefaultMaintObj : isDefaultMaintObj
  };
};

module.exports = MaintenanceServiceHelper();
