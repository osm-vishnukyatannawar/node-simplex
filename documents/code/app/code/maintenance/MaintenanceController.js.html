var MaintenanceService = require(__CONFIG__.app_code_path + 'maintenance/MaintenanceService');
var logger = require(__CONFIG__.app_base_path + 'logger');
var Controller = require(__CONFIG__.app_base_path + 'lib/controller');
var util = require("util");

function MaintenanceController(app) {
  Controller.call(this);
  this.sMaintenance = new MaintenanceService();
  app.httpPost('tag/maintenance', this.processTagPost.bind(this), true);
}

util.inherits(MaintenanceController, Controller);

MaintenanceController.prototype.processTagPost = function(request, response) {
  var tagPostObj = request.body;
  var that = this;
  this.sMaintenance.processTagPost(tagPostObj, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

module.exports = MaintenanceController;
