EmailService.prototype.sendUnsentMails = function() {
  var mEmail = new Email();
  var processResponse = function(err, emResponse) {
    if (err) { return cntrlCb(err); }

    var successArray = [];
    var errArray = [];
    i = 0;
    var finalResponse = {};
    __.each(emResponse.success, function(emailResult) {
      successArray.push(emailResult.id);
    });

    mEmail.markAsSent(successArray, function(err, data) {
      if (err) {
        finalResponse.success = err;
      } else {
        finalResponse.success = true;
      }
      checkIfDone();
    });

    __.each(emResponse.error, function(emailResult) {
      errArray.push(emailResult.id);
    });

    mEmail.markAsError(errArray, function(err, data) {
      if (err) {
        finalResponse.error = err;
      } else {
        finalResponse.error = true;
      }
      checkIfDone();
    });

    function checkIfDone() {
      ++i;
      if (i === 2) { return logFailedMails(emResponse, finalResponse); }
    }
  };
  
  mEmail.getUnsentMail(maxMailsToSend, function(err, data) {
    if (err) { return logger.writeLogErr(err); }
    mEmail.email.mailer.sendMails(data, processResponse);
  });
};
