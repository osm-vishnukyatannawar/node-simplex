var Model = require(__CONFIG__.app_base_path + 'lib/model');
var uuid = require('node-uuid');
var AppError = require(__CONFIG__.app_base_path + 'lib/app-error');
var emailProperties = require(__CONFIG__.app_code_path + 'emails/Properties');
var util = require('util');

function Email(objToBind, requestParameters) {
  Model.call(this, this.properties, objToBind, requestParameters);
}

util.inherits(Email, Model);

Email.prototype.properties = emailProperties;

Email.prototype.queueMail = function(mailObj, cb) {
  if (mailObj.data) {
    try {
      mailObj.data.email = this.getDefaultParams();
      mailObj.data = JSON.stringify(mailObj.data);
    } catch (e) {
      mailObj.data = getDefaultData();
    }
  } else {
    mailObj.data = getDefaultData();
  }
  var sqlQuery = 'INSERT INTO email_m (`email_recid`, `subject`, `to_email`, `cc_email`, `bcc_email`, `from_email`, `template_name`,'
          + '`data`, `created_on`,  `modified_on`, `is_sent`) VALUES(:id, :subject, :toEmail, :ccEmail, :bccEmail, :fromEmail, :templateName,'
          + ':data, :createdOn, :modifiedOn, :isSent);';
  this.query({
    query: sqlQuery,
    data: mailObj,
    cb: function(err, data) {
      if (err) { return cb(err); }
      return cb(null, data);
    }
  });
};

Email.prototype.getUnsentMail = function(limit, cb) {
  if (isNaN(parseInt(limit))) {
    limit = 50;
  }
  var sqlQuery = 'SELECT email_recid AS emailID, subject, to_email AS toEmail, cc_email AS ccEmail, '
          + ' from_email AS fromEmail, template_name AS templateName, data, created_on AS createdOn, modified_on AS modifiedOn'
          + ' FROM email_m WHERE is_sent = ? ORDER BY created_on ASC LIMIT '
          + limit + ';';
  this.getResults({
    query: sqlQuery,
    data: [this.getUnsentMailID()],
    cb: function(err, data) {
      return cb(err, data);
    }
  });
};

Email.prototype.markAsSent = function(emailRecIDs, cb) {
  var lstOfMails = updateMailStatus(this, emailRecIDs, cb, true)
  if(lstOfMails) {
    this.query(lstOfMails);
  }
};

Email.prototype.markAsError = function(emailRecIDs, cb) {
  var lstOfMails = updateMailStatus(this, emailRecIDs, cb, true)
  if(lstOfMails) {
    this.query(lstOfMails);
  }  
};

Email.prototype.getDefaultObj = function() {
  var currDate = this.dbUtils.toMySQLDateTime();
  return {
    createdOn: currDate,
    modifiedOn: currDate,
    isSent: 0,
    id: uuid.v4(),
    data: '',
    templateName: '',
    ccEmail: '',
    bccEmail: '',
    fromEmail: __CONFIG__.email.fromName + ' <' + __CONFIG__.email.username
            + '>'
  };
};

Email.prototype.getSentMailID = function() {
  return 1;
};

Email.prototype.getUnsentMailID = function() {
  return 0;
};

Email.prototype.getErrorMailID = function() {
  return -1;
};

Email.prototype.getDefaultParams = function() {
  return {
    baseURL: __CONFIG__.email.baseURL,
    fromName: __CONFIG__.email.fromName
  };
};

function getDefaultData() {
  var mailObj = {};
  mailObj.data = {};
  mailObj.data.email = this.getDefaultParams();
  mailObj.data = JSON.stringify(mailObj.data);
  return mailObj.data;
}

function updateMailStatus(emailObj, emailRecIDs, cb, isError) {
  if (emailRecIDs && emailRecIDs.length === 0) { return false; }
  var questionMarks = [];
  var finalData = [];
  if (isError) {
    finalData.push(emailObj.getErrorMailID());
  } else {
    finalData.push(emailObj.getSentMailID());
  }

  for (var i = 0, len = emailRecIDs.length; i < len; ++i) {
    if (emailObj.validator.isUUID4(emailRecIDs[i])) {
      questionMarks.push('?');
      finalData.push(emailRecIDs[i]);
    }
  }
  questionMarks = questionMarks.join();
  var sqlQuery = 'UPDATE email_m SET is_sent = ? WHERE email_recid IN ('
          + questionMarks + ');';
  return {
    query: sqlQuery,
    data: finalData,
    validate: false,
    cb: function(err, data) {
      return cb(err, data);
    }
  };
}

module.exports = Email;
