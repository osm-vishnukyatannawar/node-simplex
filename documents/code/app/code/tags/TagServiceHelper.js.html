¶ Format Tag configuration details

Formats the tag configuration object, groups the details.

Params
data {object} tag configuration object to be formatted returns formatted tag configuration object 
  var formatTagConfigDetails = function(data) {
    var formattedData = {};
    var wifiParams    = ['serverIPAddr','numDNS_Servers','DNS_IP_Addr1','DNS_IP_Addr2','networkName',
                         'networkSSID','passphrase','networkSecurity','addrtype','tagStaticIPAddr'];
    var bleElements   = ['bleEnable','bleGapRoleMinConnIntervalms','bleGapRoleMaxConnIntervalms','bleGapRoleSlaveLatency',
                         'bleLimitedAdvertOnTimeSec','bleLimitedAdvertIntervalTimems','bleGeneralAdvertIntervalTimems',
                         'bleLowBattAdvertIntervalTimems','bleDesiredConnectionTimeoutms','advertOffTimems','blePasskey'];
    var devParam    = ['buzzerFreqHz','buzzerOnTime_msec','buzzerOffTime_msec','buzzerCnt','ledOnTime_msec',
                       'ledOffTime_msec','ledCnt'];
    var currentValues   = ['threshOff','threshInact','threshActive'];
    var histValues    = ['histResetRule','histThresh'];
    var pimValues     = ['elapsedTime_10ms','iirFilt_mHz','rmsDiffThresh_mA','trigDly_10ms',
                          'sampRateHz','measDur_10ms'];
    var batteryLevels   = ['batteryRechargeLevelPercent','batteryChargeLevelPercent','batteryThreshLowAlert','batteryThreshCriticalAlert'];
    
    
    formattedData = data.main;
    delete formattedData.tagConfigParamsRecId;
    
    formattedData.wifiNetworkParams = {};
    for (var i = 0; i < wifiParams.length; i++) { 
        formattedData.wifiNetworkParams[wifiParams[i]] = data.secondary[wifiParams[i]];
      }  
    
    formattedData.bleConfig = {};
    for (var i = 0; i < bleElements.length; i++) {  
        formattedData.bleConfig[bleElements[i]] = data.secondary[bleElements[i]];
      }
    
    formattedData.devFindParam = {};
    for (var i = 0; i < devParam.length; i++) { 
        formattedData.devFindParam[devParam[i]] = data.secondary[devParam[i]];
      }
    
    formattedData.currentMeasThresh = {};
    for (var i = 0; i < currentValues.length; i++) {  
        formattedData.currentMeasThresh[currentValues[i]] = data.secondary[currentValues[i]];
      }
    
    formattedData.histConfig = {};
    formattedData.histConfig.enable = data.secondary.histEnable;
    for (var i = 0; i < histValues.length; i++) { 
        formattedData.histConfig[histValues[i]] = data.secondary[histValues[i]];
      }
    
    formattedData.pimConfig = {};
    formattedData.pimConfig.enable = data.secondary.pimEnable;
    formattedData.pimConfig.trigger = data.secondary.pimTrigger;
    for (var i = 0; i < pimValues.length; i++) {  
        formattedData.pimConfig[pimValues[i]] = data.secondary[pimValues[i]];
      }
    
    formattedData.batteryConfig = {};
    for (var i = 0; i < batteryLevels.length; i++) {  
        formattedData.batteryConfig[batteryLevels[i]] = data.secondary[batteryLevels[i]];
      }
      
     return formattedData;
  };
  
  return {
    formatTagConfigDetails : formatTagConfigDetails   
  };
};
