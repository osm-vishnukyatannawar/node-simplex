User.prototype.getUserSelectQuery = function(whereClause, isOrgID) {
  if (__.isEmpty(whereClause)) {    
    whereClause = '';
  }
  var innerWhereClause = '';
  var outerWhereClause = '';
  extendedSelectCols = selectCols + ' , roles_det.deployments_raw ';

  var sqlQuery = extendedSelectCols + fromTable + ' ' +
    'INNER JOIN (SELECT DISTINCT urd.user_recid,' +
    'GROUP_CONCAT(CONCAT(dm.deployment_recid, \'~|\', dm.deployment_name, \'~|\', urd.organization_recid,\'~|\', urd.role_recid) ' +
    'SEPARATOR \';|\') deployments_raw ' + 'FROM user_role_det AS urd  ' +
    'INNER JOIN role_m AS rm ON rm.role_recid = urd.role_recid ' +
    'INNER JOIN deployment_m AS dm ON dm.deployment_recid = urd.deployment_recid ' +
    'AND dm.organization_recid = urd.organization_recid ' +
    'WHERE urd.is_active = 1 AND rm.is_active = 1 AND dm.is_active = 1 %s ' +
    'GROUP BY urd.user_recid) roles_det ' + 'ON roles_det.user_recid = u.user_recid ' +
    'WHERE %s u.is_active = 1';
  
  if(whereClause.length !== 0) {
    if(isOrgID) {
      innerWhereClause = 'AND ' + whereClause;
    } else {
      outerWhereClause = whereClause + ' AND';    
    }
  }
  
  sqlQuery = util.format(sqlQuery, innerWhereClause, outerWhereClause);
  sqlQuery += ';';

User.prototype.checkIfBelongsToOrg = function(userInfo, orgID, cb) {
  var sqlCondition = '';
  if (__.isEmpty(userInfo)) {

      .getStatusCode('badRequest'),
      'Please send the user email or user ID.', {}));
  }
  var userData = '';
  if (userInfo.emailID && this.validator.isEmail(userInfo.emailID)) {
    sqlCondition = 'um.email_id = ?';
    userData = userInfo.emailID;
  } else if (userInfo.userID && this.validator.isUUID4(userInfo.userID)) {
    sqlCondition = 'um.user_recid = ?';
    userData = userInfo.userID;
  } else {
    return cb(new AppError(this.getStatusCode('badRequest'),
      'Please send the user email or user ID.', {}));
  }

  if (__.isEmpty(orgID)) {
    return cb(new AppError(this
      .getStatusCode('badRequest'), 'Please provide the organization ID.', {}));
  }
  sqlCondition += ' AND om.organization_recid = ?';
  var sqlQuery = this.getOrgSelectQuery(sqlCondition);
  var that = this;
  this.getResult({
    query: sqlQuery,
    data: [userData, orgID],
    cb: function(err, data) {
      if (err) {
        return cb(err);
      }
      if (__.isEmpty(data)) {
        return cb(new AppError(that.getStatusCode('notFound'),
          'No user found with emailID/userID - ' + userInfo + ' in organization.', {}));
      } else {
