var UserService = require(__CONFIG__.app_code_path + 'users/UserService');
var logger = require(__CONFIG__.app_base_path + 'logger');
var Controller = require(__CONFIG__.app_base_path + 'lib/controller');
var util = require('util');

function UserController(app) {
  Controller.call(this);
  app.post(__CONFIG__.app_base_url + 'user/login', this.loginUser.bind(this));
  app.post(__CONFIG__.app_base_url_token + 'user/logout', this.logoutUser.bind(this));
  app.get(__CONFIG__.app_base_url + 'user/exists', this.checkIfUserExists.bind(this));
  app.post(__CONFIG__.app_base_url + 'user/register', this.registerUser.bind(this));
  app.post(__CONFIG__.app_base_url + 'user/', this.addUser.bind(this));
  app.httpGet('organization/:orgID/users', this.getUserByOrgID.bind(this));
  app.httpPost('user/add', this.addUser.bind(this));
}

util.inherits(UserController, Controller);

UserController.prototype.checkIfUserExists = function(request, response) {  
  var emailID = request.query.emailID;
  var that = this;  
  var sUser = new UserService(request.queryParams);
  sUser.checkIfUserExists(emailID, function(err, data) {
    that.sendResponse(err, data, response);
  });  
};

UserController.prototype.loginUser = function(request, response) {
  var that = this;
  var sUser = new UserService(request.queryParams);
  sUser.loginUser(request.body, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

UserController.prototype.logoutUser = function(request, response) {
  var that = this;
  var sUser = new UserService(request.queryParams);
  sUser.logoutUser(request.body.userID, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

UserController.prototype.registerUser = function(request, response) {
  var that = this;
  var sUser = new UserService(request.queryParams);
  sUser.registerUser(request.body, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

UserController.prototype.addUser = function(request, response) {
  var that = this;
  var sUser = new UserService();
  sUser.addUser(request.body, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

UserController.prototype.getUserByOrgID = function(request, response) {
  var orgID = request.params.orgID;
  var that = this;
  var sUser = new UserService();
  sUser.getUsersByOrg(orgID, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

module.exports = UserController;
