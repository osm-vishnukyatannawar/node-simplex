var UserServiceHelper = function() {
  var colSplit = '~|';
  var rowSplit = ';|';
  var processSelectResponse = function(data) {
    if (!data) {
      return data;
    }
    if (__.isArray(data)) {
      for (var i = 0, len = data.length; i != len; ++i) {
        data[i] = getUserObj(data[i]);
      }
    } else {
      data = getUserObj(data);
    }
    return data;
  };
  
  var getUserObj = function(data) {
    data['deployments'] = getDeployments(data['deployments_raw']);
    delete data['deployments_raw'];
    return data;
  };
  
  var getDeployments = function(colData) {
    if (!colData) {
      return {};
    }
    var deployments = colData.split(rowSplit);
    var respArr = [];
    for (var i = 0, len = deployments.length; i != len; ++i) {
      var deployment = deployments[i].split(colSplit);
      if (deployment.length === 4) {
        respArr.push({
          "id": deployment[0],
          "name": deployment[1],
          "organizationID": deployment[2],
          "roleID": deployment[3]
        });
      }
    }
    return respArr;
  };
  
  var getOrganizations = function(userData, data) {
    var organizations = data;
    userData['organizations'] = [];
    for (var i = 0, len = organizations.length; i != len; ++i) {
      userData['organizations'].push({
        'id': organizations[i].organizationID,
        'name': organizations[i].organizationName,
      });
    }
    return userData;
  };   
  
  return {
    getUserObj : getUserObj,
    getOrganizations : getOrganizations,
    processSelectResponse : processSelectResponse   
  };
};
