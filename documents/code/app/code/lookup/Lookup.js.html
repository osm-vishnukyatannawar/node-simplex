Lookup.prototype.getValuesFromLookup = function(propertyName , cb){
  var isActive = 1;
  var sqlQuery = "SELECT * FROM tag_lookup_table WHERE Lookup_Key = ? AND Is_Active = ?"
  this.getResults({
      query: sqlQuery,
      data: [propertyName , isActive],
      cb: function(err, queryResult) {
        if (!err && __.isEmpty(queryResult)) {
          return cb(null, false);
        }
        if (err) {
         
          return cb(new AppError(that.getStatusCode("internalError"),
            "There was a problem on the server. Please try again later", err));
        }
        return cb(null, queryResult);
      }
    });
}
