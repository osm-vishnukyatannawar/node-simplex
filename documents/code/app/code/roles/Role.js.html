var Model = require(__CONFIG__.app_base_path + 'lib/model');
var uuid = require('node-uuid');
var __ = require('underscore');
var AppError = require(__CONFIG__.app_base_path + 'lib/app-error');
var util = require('util');

function Role(objToBind, requestParameters) {
  Model.call(this, this.properties, objToBind, requestParameters);
}

util.inherits(Role, Model);

Role.prototype.getAll = function(cb) {
  var requestParams = this.queryModifiers;
  var that = this;
  var sqlQuery = ("SELECT role_m.role_recID AS roleID, role_m.role_name AS roleName, "
          + "role_m.created_on AS createdOn,  role_m.modified_on AS modifiedOn "
          + "FROM role_m " + "WHERE role_m.is_active = ?");
  this.getResults({
    query: sqlQuery,
    data: [this.getActiveID()],
    cb: function(err, data) {
      if (!err && __.isEmpty(data)) {
        err = new AppError(that.getStatusCode('notFound'),
                'No roles were defined.', {});
      }
      cb(err, data);
    }
  });
};

Role.prototype.getFullRoleID = function(cb) {  
  var sqlQuery = ('SELECT role_m.role_recID AS roleID FROM role_m WHERE role_m.is_active = ? AND role_m.role_name = "Full"');
  this.getResult({
    query: sqlQuery,
    data: [this.getActiveID()],    
    cb: function(err, data) {
      if (!err && __.isEmpty(data)) {
        err = new AppError(that.getStatusCode('notFound'),
                'A role with that name was not found.', {});
      }
      if (err) { return cb(err); }
      cb(null, { roleID : data.roleID });
    }
  });
};

module.exports = Role;
