DeploymentController.prototype.getDeploymentByID = function(request, response) {
  var that = this;
  var id = request.params.id;
  this.Deployment = new Deployment();
  this.Deployment.getDeploymentByID(id, function(err, data) {
    that.sendResponse(err, data, response);
  });
};

module.exports = DeploymentController;
