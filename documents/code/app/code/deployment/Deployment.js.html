  var sqlQuery = ("SELECT deployment_m.deployment_recid AS deploymentID, deployment_m.deployment_name AS deploymentName, "

          + "FROM deployment_m "
          + "WHERE deployment_m.deployment_recid = ? AND deployment_m.is_active = ?");
  this.getResult({
    query: sqlQuery,
    data: [deploymentID, this.getActiveID()],
