ReportProblem.prototype.insertReport = function(reportObj, cb) {
  var sqlQuery = 'INSERT INTO report_problem_m (report_problem_recid, user_recid, organization_recid, deployment_recid,'
        + 'description, filepath, filename, is_active,'
          + 'created_on,  modified_on) VALUES(:id, :userID, :organizationID, :deploymentID,'
          + ':description, :filepath, :filename, :isActive,'
          + ':createdOn, :modifiedOn);';
  this.query({
    query: sqlQuery,
    data: reportObj,
    cb: function(err, data) {
      if(err) {
        return cb(err,{});
      }
      return cb(null, {
        reportDataID : reportObj.id
      });
   }  
  });
};

ReportProblem.prototype.getDefaultReportData = function() {
  var currDate = this.dbUtils.toMySQLDateTime();  
  
  return {
    id: uuid.v4(),
    createdOn: currDate,
    modifiedOn: currDate,
    isActive: this.getActiveID()
  };
};
