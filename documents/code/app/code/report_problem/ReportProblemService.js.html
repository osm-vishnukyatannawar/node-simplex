ReportProblemService.prototype.reportProblem = function(clientData, cntrlCb) {

  var mReport = new ReportProblem();

  async.waterfall([
    function(next) {
      var reportData = clientData;
      reportData = __.defaults(reportData, mReport.getDefaultReportData());
      mReport.insertReport(reportData, next);
    },
    function(reportDataID, next) {
      var ReportProblemInfo = {
        username: (clientData.username).trim(),
        organization: (clientData.organization).trim(),
        deployment: (clientData.deployment).trim(),
        reportDescription: (clientData.description).trim(),
        dateAndTime: (clientData.dateAndTime).trim()
      };

      var email = new Email();
      var dataToSave = {
        ReportProblemData: {
          username: ReportProblemInfo.username,
          organization: ReportProblemInfo.organization,
          deployment: ReportProblemInfo.deployment,
          reportDescription: ReportProblemInfo.reportDescription,
          dateAndTime: ReportProblemInfo.dateAndTime
        }
      };

      var files = __.values(clientData.files);
      var attachments = [];
      if(files && files.length > 0) {
        for(var i = 0; i < files.length; ++i ) {
            attachments.push({
              filename: files[i].name,
              path: files[i].path
            });
          }
      }

      email.queueMail(__.extend(email.getDefaultObj(), {
        toEmail: global.__CONFIG__.email.emailsToSend,
        templateName: 'report-problem',
        subject: 'Problem reported on Emanate portal',
        data: dataToSave,
        attachments: attachments
      }), function(err, data) {
        if (err) {
          err = new AppError(err, 'There was an error while sending the email to the administrator.', {});
        }
        next(err);
      });
    }
  ], function(err, data) {
    if (err) {
      return cntrlCb(new AppError(err,
        'There was an error while reporting a problem.', {}));
    }
    var emailService = new EmailService();
    emailService.sendUnsentMails();
    return cntrlCb(null, {});
  });
};
