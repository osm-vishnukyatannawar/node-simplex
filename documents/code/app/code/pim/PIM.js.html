PIM.prototype.insertPIMHeader = function(pimHeader, cb) {
  var transactionID = null;
  if (pimHeader.hasOwnProperty('transactionID')) {
    transactionID = pimHeader.transactionID;
    pimHeader = pimHeader.data;
  }
  var sqlQuery = 'INSERT INTO pim_stats_header(`pim_stats_header_recid`, `tag_srno`,  `num_ac_plugins`, `num_pim_blocks`,'
          + '`num_samples_in_block`,`created_on`,`created_by`, `organization_id`, `cassandra_dump_rec_id`)'
          + ' VALUES(:id, :tagSN, :numACPlugins, :numPimBlocks, :numSamplesInBlock, :createdOn, :createdBy, :organizationID, :dmpRecID);';
  this.query({
    query: sqlQuery,
    data: pimHeader,
    transactionID: transactionID,
    cb: function(err, data) {
      if (err) { return cb(err); }
      return cb(null, {
        pimHeaderID: pimHeader.id
      });
    }
  });
};

PIM.prototype.insertPIMData = function(pimData, cb) {
  var transactionID = null;
  if (pimData.hasOwnProperty('transactionID')) {
    transactionID = pimData.transactionID;
    pimData = pimData.data;
  }
  var queryObj = this.getPimDataMappingQuery(pimData);
  this.query({
    query: queryObj.query,
    data: queryObj.data,
    transactionID: transactionID,
    cb: function(err, data) {
      if (err) { return cb(err); }
      if (pimData.length === data.affectedRows) { return cb(null, {
        success: true
      }); }
      return cb(null, {
        success: false
      });
    }
  });
};

PIM.prototype.getPimDataMappingQuery = function(pimData) {
  for (var i = 0, len = pimData.length; i < len; ++i) {
    pimData[i].plugInTimeStamp = this
            .convertTagTimeStamp(pimData[i].plugInTimeStamp);
  }
  var sqlQuery = 'INSERT INTO pim_stats_data(`pim_stats_data_recid`, `tag_pim_trig_id_enum`, '
          + '`plugin_timestamp`,  `plugin_dur_min`, `plugin_to_measure_delay_msec`,'
          + '`rms_current`,`peak_to_rms`,`std_dev`, `pim_stats_header_recid`, `created_by`,`created_on`) VALUES';
  var currDate = this.dbUtils.toMySQLDateTime();
  var finalData = {
    pimHeaderID: pimData.pimHeaderID,
    createdOn: currDate,
    createdBy: __CONFIG__.user.default_created_by
  };
  var propNames = ['trig', 'plugInTimeStamp', 'plugInDur_min',
      'plugInToMeasDelay_msec', 'rmsCurrent', 'peakToRms', 'stdDev'];
  var queryObj = this.getMultipleInsertQuery({
    initialQuery: sqlQuery,
    staticData: finalData,
    propNames: propNames,
    staticCols: [':pimHeaderID', ':createdBy', ':createdOn'],
    data: pimData,
    hasPrimary: true
  });

  return queryObj;
};

PIM.prototype.getAllPIMData = function(cb) {
  var sqlQuery = 'SELECT head.tag_srno AS tagSN, head.organization_id AS organizationID, head.curr_timestamp AS currTimeStamp, '
          + 'head.num_ac_plugins AS numACPlugins, head.num_pim_blocks AS numPIMBlocks, head.num_samples_in_block AS numSamplesInBlock, '
          + '(SELECT lookup_value_vc FROM tag_lookup_table WHERE lookup_key = \'powerPathPimTrig\' AND '
          + 'lookup_value_id = stats.tag_pim_trig_id_enum) trig, stats.plugin_dur_min AS pluginDurMin, '
          + 'stats.plugin_to_measure_delay_msec AS pluginToMeasureDelay, stats.rms_current AS rmsCurrent, '
          + 'stats.peak_to_rms AS peakToRms, stats.std_dev AS stdDev FROM pim_stats_data stats '
          + ' INNER JOIN (SELECT * FROM pim_stats_header head ORDER BY created_on DESC) '
          + ' head ON stats.pim_stats_header_recid = head.pim_stats_header_recid;';
  this.getResults({
    query : sqlQuery,
    data : [],
    cb : cb
  });
};
