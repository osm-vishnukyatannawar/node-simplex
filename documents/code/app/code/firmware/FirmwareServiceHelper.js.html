FirmwareServiceHelper.prototype.extactFilesTofolder = function(fwUploadPropObj, cb) {
  var version = fwUploadPropObj.clientObj.version.trim();
  var sourcePath = fwUploadPropObj.clientObj.path.trim();
  var filesFolder = global.__CONFIG__.filesFolderName;
  var firmwareFolder = global.__CONFIG__.firmware.folder;
  var destPath = global.__CONFIG__.app_base_path + '../' + filesFolder + '/' + firmwareFolder + '/' +
    global.__CONFIG__.firmware.baseVersionFolderName +
    version;
  var sService = new Service();
  sService.UnzipFilesTofolder(sourcePath, destPath, function(err) {
    if (err) {
      return cb(err);
    }
    cb(null, fwUploadPropObj);
  });

};

FirmwareServiceHelper.prototype.getPropertiesArray = function(fileNameArray, version, cb) {
  var that = this;
  var isValid = true;
  var errorMsg;
  var propsArray = {};
  for (var i = 0; i < fileNameArray.length; i++) {
    var element = fileNameArray[i];
    var extension = element.substr(element.length - 4);
    if (extension !== global.__CONFIG__.firmwareFileExtension) {

      errorMsg = "Invalid file type. Please check and upload again";
      return cb(new AppError(that.getStatusCode('badRequest'),
        errorMsg, {}));
    };
    var validFileName = false;
    try {
      var DefaultObj = global.__CONFIG__.firmware.baseFirmwareFileStrs;
      for (var firmwareElementKey in DefaultObj) {
        if (DefaultObj.hasOwnProperty(firmwareElementKey)) {
          var firmwareElement = DefaultObj[firmwareElementKey];
          var elementPos = element.search(firmwareElement);
          if (elementPos > -1) {
            propsArray[firmwareElement] = {};
            validFileName = true;
            var res = element.split(firmwareElement);
            var extension = res[1].split(global.__CONFIG__.firmwareFileExtension);
            propsArray[firmwareElement]['version'] = extension[0];
            propsArray[firmwareElement]['name'] = element;
            propsArray[firmwareElement]['path'] = global.__CONFIG__.firmware.baseVersionFolderName +
              version + '/' +
              element;
          };
        };
      };
    } catch (e) {

      return cb(new AppError(that.getStatusCode('internalError'),
        "There was an error on the server.Please try again later.", e));
    }
    if (!validFileName) {
      errorMsg = "Invalid files in the zip file.";
      return cb(new AppError(getStatus("badRequest"),
        errorMsg, {}));
    };
  };
  return cb(null, propsArray);
};
