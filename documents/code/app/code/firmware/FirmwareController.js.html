FirmwareController.prototype.uploadFirmware = function(request, response) {
  var that = this;
  var path = request.files.uploadFile.path;
  var userId = request.currToken.userID;
  var version = request.fields.version;
  var changeLog = request.fields.changeLog;
  var obj = {
    'path': path,
    'version': version,
    'changeLog': changeLog,
    'userId': userId
  };
  this.sFirmware.uploadFirmwareFiles(obj, function(err, data) {
    that.sendResponse(err, data, response);
  });
};
