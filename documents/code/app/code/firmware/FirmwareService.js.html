This method is used to upload the firmware files It follows async waterfall model. It takes the folder names and stuff from the global.__CONFIG__.firmware

Flow of the function

  1. It will get all the details of the files in the zip file.
  2. It will check if the version exists in the database or not.(Version should be unique)
  3. It will split the details of that we got in the 1st step and extracts versions numbers of each file and name of each file
  4. Next it will extract all the files to the destination folder
  5. Next it will check if all the three files exists in the zip file if not it will take the latest firmware version and move it to the destination folder.
  6. Next it will insert all the details into the database.
Params
: dataObj -This is contains path , version , changeLog , userId
: cb -Callback 
FirmwareService.prototype.uploadFirmwareFiles = function(dataObj, cb) {
  var that = this;
  var path = dataObj.path;
  var version = dataObj.version;
  var changeLog = dataObj.version;
  this.getDetailsFromZip(path, function(err, data) {
    if (err) {
      return cb(err);
    }

    var NoOfFiles = data.length;
    var fileNameArray = [];
    for (var i = 0; i < NoOfFiles; i++) {
      var firmwareName = data[i].entryName;
      fileNameArray.push(firmwareName);
    }

    async.waterfall([
        function(wfcb) {
          that.checkIfVersionExists(version, wfcb);
        },
        function(wfCb) {
          that.shFirmwareServiceHelper.getPropertiesArray(fileNameArray, version, wfCb);
        },
        function(propArray, wfCb) {
          that.shFirmwareServiceHelper.extactFilesTofolder({
            'propArray': propArray,
            'clientObj': dataObj
          }, wfCb);
        },
        function(FirmwareObj, wfCb) {
          that.getAllFilesForZip(FirmwareObj, wfCb);
        },
        function(FirmwareObj, wfCb) {
          that.insertFirmwareToDb(FirmwareObj, wfCb);
        }
      ],
      function(err, data) {
        if (err) {
          return cb(err);
        }
        cb(null, data);
      });

  });
};

FirmwareService.prototype.getAllFilesForZip = function(FirmwareObjMethod, cb) {
  var completObj = FirmwareObjMethod;
  var propObj = FirmwareObjMethod.propArray;
  var clientObj = FirmwareObjMethod.clientObj;
  var propLength = __.size(FirmwareObjMethod.propArray);
  if (propLength === 3) {
    return cb(null, FirmwareObjMethod);
  }
  this.mFirmware.getLatestFirmware(function(err, data) {
    if (err) {
      return cb(err);
    }
    var overallVersionOld = data.overall_version;
    var filesFolder = global.__CONFIG__.filesFolderName;
    var firmwareFolder = global.__CONFIG__.firmware.folder;
    var oldPath = global.__CONFIG__.app_base_path + "../" + filesFolder + "/" + firmwareFolder + "/" +
      global.__CONFIG__.firmware.baseVersionFolderName +
      overallVersionOld + "/";
    var newPath = global.__CONFIG__.app_base_path + "../" + filesFolder + "/" + firmwareFolder + "/" +
      global.__CONFIG__.firmware.baseVersionFolderName +
      clientObj.version + "/";
    var globalFirmwareObj = global.__CONFIG__.firmware.baseFirmwareFileStrs;
    for (var key in globalFirmwareObj) {
      if (propObj.hasOwnProperty(globalFirmwareObj[key])) {
        continue;
      } else {
        var oldVersion = data[key];
        var oldFolderName = globalFirmwareObj[key] + oldVersion;
        var oldPathFolderName = oldPath + oldFolderName + '.zip';
        var newPathFolderName = newPath + oldFolderName + '.zip';
        try {
          fse.copy(oldPathFolderName, newPathFolderName, function(err) {
            if (err) {
              console.log(err);
              return cb(new AppError(getStatus("internalError"),
                "There was an error on the server while processing the firmwares.", {}));
            }
            var firmwareName = globalFirmwareObj[key] + oldVersion + '.zip';
            FirmwareObjMethod.propArray[globalFirmwareObj[key]] = {};
            FirmwareObjMethod.propArray[globalFirmwareObj[key]]['name'] = firmwareName;
            FirmwareObjMethod.propArray[globalFirmwareObj[key]]['version'] = oldVersion;
            FirmwareObjMethod.propArray[globalFirmwareObj[key]]['path'] = global.__CONFIG__.firmware
              .baseVersionFolderName +
              clientObj.version + "/" + firmwareName;
            cb(null, FirmwareObjMethod)
          });
        } catch (e) {
          return cb(new AppError(getStatus("internalError"),
            "There was an error on the server while processing the firmwares.", e));
        }
      }
    }

  });
};

FirmwareService.prototype.normalizeFirmwareData = function(dataToNormalize, cb) {
  var that = this;
  var baseFirmwareFolder = global.__CONFIG__.firmware.folder;
  var totalVersionFolderName = global.__CONFIG__.firmware.baseVersionFolderName;
  var filesFolderName = global.__CONFIG__.filesFolderName;
  var baseURL = global.__CONFIG__.email.baseURL;
  var fileExt = global.__CONFIG__.firmware.firmwareFileExtension;
  var finalURL = baseURL + filesFolderName + '/' + baseFirmwareFolder + '/' + totalVersionFolderName;
  for (var element in dataToNormalize) {
    var overAllVersion = dataToNormalize[element]['overAllVersion'];
    dataToNormalize[element]['hostURL'] = finalURL + overAllVersion + '/' + dataToNormalize[element][
      'hostPath'
    ];
    dataToNormalize[element]['bleURL'] = finalURL + overAllVersion + '/' + dataToNormalize[element][
      'blePath'
    ];
    dataToNormalize[element]['wifiURL'] = finalURL + overAllVersion + '/' + dataToNormalize[element][
      'wifiPath'
    ];
    dataToNormalize[element]['overAllURL'] = finalURL + overAllVersion;
    delete dataToNormalize[element].hostPath;
    delete dataToNormalize[element].blePath;
    delete dataToNormalize[element].wifiPath;
  }

  cb(null, dataToNormalize);
};
