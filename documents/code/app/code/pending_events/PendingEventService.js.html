function checkIfTagIsValid(eventObj, next) {
  var sTag = new tagService();
  sTag.checkIfTagBelongsToOrg(eventObj.tagSN, eventObj.organizationID, function(err, data) {
    if (err) {
      return next(err);
    }
    if (data.belongsToOrg && data.exists) {
      return next(null);
    } else {
      return next(sTag.getStatusCode('badRequest', 'The tag and organization ID are not valid.', {}));
    }
  });
