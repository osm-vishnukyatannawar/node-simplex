PendingEvent.prototype.getPendingEvents = function(serialNumber, cb) {
  if (__.isEmpty(serialNumber)) {
    return cb(new AppError(this
      .getStatusCode('badRequest'),
      'Please send the tag\'s serial number.', {}));
  }

  var sqlQuery =
    'SELECT pending_event_recid AS id , event_type_id_enum AS eventTypeID, DATA AS data FROM pending_events' +
    ' WHERE tag_srno = ? AND is_active = ? AND is_processed = ?';

  this.getResults({
    query: sqlQuery,
    data: [serialNumber, this.getActiveID(), this.getUnprocessedFlag()],
    cb: function(err, data) {
      cb(err, data);
    }
  });
};

PendingEvent.prototype.setProcessedFlag = function(tagSN, cb) {
  if (__.isEmpty(tagSN)) {
      return cb(new AppError(this
        .getStatusCode('badRequest'),
        'Please send the tag\'s serial number.', {}));
    }
  
  var sqlQuery = 'UPDATE pending_events SET is_processed =  ?' +
    ' WHERE tag_srno = ?  AND is_active = ? AND is_processed = ?';

  this.query({
    query: sqlQuery,
    data: [this.getProcessedFlag(), tagSN, this.getActiveID(),
           this.getUnprocessedFlag()],
    validate : false,
    cb: function(err, data) {      
      return cb(err, data);
    }
  });
};

PendingEvent.prototype.getDefaultObj = function() {
  var currDate = this.dbUtils.toMySQLDateTime();
  return {
    createdOn : currDate,
    isActive : this.getActiveID(),
    isProcessed : this.getUnprocessedFlag(),
    data : '',
    id : uuid.v4(),
    createdBy : __CONFIG__.user.default_created_by
  };
};

PendingEvent.prototype.getUnprocessedFlag = function() {
  return 0;
};

PendingEvent.prototype.getProcessedFlag = function() {
  return 1;
};

PendingEvent.prototype.getLookupKey = function() {
  return 'powerPathCmds';
