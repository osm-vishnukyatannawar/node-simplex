Log.prototype.insertLog = function(logObj, cb) {

  if (!this.validator.isUUID4(logObj.userID) || __.isEmpty(logObj.logType)) {
    return cb(new AppError(this.getStatusCode('badRequest'), i18n.__('Please send all necessary parameters.'), 
        {}));
  }
  var sqlQuery = 'INSERT INTO log_m (log_recid, user_id, log_type, '
        + 'log_description, is_active, created_on) '
        + 'VALUES(:logID, :userID, :logType,'
          + ':logDescription, :isActive, :createdOn);';
  this.query({
    query: sqlQuery,
    data: logObj,
    cb: function(err, data) {
      if(err) {
        return cb(err,{});
      }
      return cb(null, {
        logID : logObj.logID
      });
   }  
  });
};

Log.prototype.filterLog = function(logType, cb) {
  var sqlQuery = 'SELECT um.user_name AS username, lm.log_recid AS logID, lm.created_on AS createdOn, '
          + 'lm.log_type AS logType, lm.log_description AS logDescription '
          + 'FROM log_m AS lm INNER JOIN user_m AS um ON lm.user_id = um.user_recid '
          + 'WHERE lm.log_type = ? AND lm.is_active = ?;';
  this.getResults({
    query: sqlQuery,
    data: [logType ,this.getActiveID()],
    cb: function(err, data) {
      if(err) {
        return cb(err,data);
      }
      return cb(err, data);
   }  
  });
};

Log.prototype.getDefaultLogData = function() {
  var currDate = this.dbUtils.toMySQLDateTime();
  return {
  logID: uuid.v4(),
    createdOn: currDate,
    isActive: this.getActiveID()
  };
