Histogram.prototype.insertHistogramData = function(histogramData, cb) { 
  var sqlQuery = 'INSERT INTO emanate.tag_hist_data (tag_hist_data_recid,cassandra_dump_rec_id, tag_srno, ' 
    + 'organization_id, tag_hist_json, created_on, created_by)'
    + ' VALUES(:id, :dmpRecID,:tagSN, :organizationID, :histData, :createdOn, :createdBy);';  
  this.query({
    query: sqlQuery,  
    data : [histogramData.id, histogramData.dmpRecID,histogramData.tagSN,histogramData.organizationID,histogramData.strData,
            histogramData.createdOn,histogramData.createdBy],
    isCassandra : true,
    cb: function(err, data) {
      if(err) {
        return cb(err,{});
      }
      return cb(null, {
        histogramID : histogramData.id
      });
    }  
  });
};

Histogram.prototype.getHistogramData = function(cb) {
  var sqlQuery = 'SELECT tag_hist_data_recid AS histID, cassandra_dump_rec_id AS dmpRecID, tag_srno AS tagSN,' + 
    'organization_id AS organizationID,  start_timestamp AS startTimestamp, hist_data AS histData, created_on AS createdOn,' +
    ' created_by AS createdBy FROM tag_hist_data;';
  this.getResults({
    query : sqlQuery,
    data : {},
    cb : cb
  });
};

Histogram.prototype.getDefaultHistogramData = function() {
  var currDate = new Date().getTime();
  return {
    id: uuid.v4(),
    createdOn: currDate,    
    dmpRecID : '',
    organizationID : '',
    tagSN : '',
    createdBy : __CONFIG__.user.default_created_by
  };
};
