Authorization.prototype.checkValidity = function(token, cb) {
  if (__.isEmpty(token)) {
    return cb(new AppError(this.getStatus('noPermission'), 'Invalid token sent.', {
      tokenID: 'The token is invalid.'
    }));    
  }
  var that = this;
  var sqlQuery = 'SELECT user_id AS userID, expires_on AS expiresOn, is_admin AS isAdmin FROM token_m WHERE token_string = ? AND is_active = ?';
  this.getResult({
    query: sqlQuery,
    data: [token, this.getActiveID()],
    cb: function(err, tokenData) {
      if (!err && __.isEmpty(tokenData)) {
        err = new AppError(that.getStatusCode('noPermission'),
          'Invalid token ' + token, {
            'token': 'Invalid token'
          });
      }
      if (err) {
        cb(err);
        return;
      }
      var currDate = that.dbUtils.toMySQLDateTime();
      if (currDate > tokenData.expiresOn) {
        cb(new AppError(that.getStatusCode('noPermission'), 'Token has expired.', {
          'token': 'Token has expired'
        }));
        that.invalidateToken(tokenData.userID);
      } else {
        cb(null, tokenData);
      }
    }
  });
};

Authorization.prototype.createNewToken = function(userID, isAdmin, cb) {
  var that = this;
  crypto.randomBytes(cryptoLength / 2, function(err, buf) {
    if (err) {
      cb(new AppError(this.getStatus('internalError'), 'Could not generate token', {}));
      return;
    }
    if(isAdmin) {
      isAdmin = 1;
    } else {
      isAdmin = 0;
    }
    var token = buf.toString('hex');
    var sqlQuery = 'INSERT INTO token_m (token_recid, user_id, token_string, created_on, expires_on, is_admin) ' +
      'VALUES (:tokenID, :userID, :token, :tokenTime, :expiryDate, :isAdmin)';
    var currDate = that.dbUtils.toMySQLDateTime();
    var expiryDate = that.dbUtils.toMySQLDateTime(that.dbUtils.addHoursToDate(null, tokenDuration));
    var dataToSend = {
      'tokenID': uuid.v4(),
      'userID': userID,
      'token': token,
      'tokenTime': currDate,
      'expiryDate': expiryDate,
      'isAdmin' : isAdmin
    };
    that.query({
      query: sqlQuery,
      data: dataToSend,
      before: that.beforeInsertUpdate(),
      cb: function(err, data) {
        if (err) {
          cb(err);
        } else {
          cb(null, token);
        }
      }
    });
  });
};


Authorization.prototype.invalidateToken = function(userID, cb) {
  if (!this.validator.isUUID4(userID)) {
    return cb(new AppError(this.getStatusCode('badRequest'), 'Please provide a valid User ID.', {
      userID: 'Please provide a valid User ID.'
    }));
  }
  var sqlQuery = 'UPDATE token_m SET is_active = ? WHERE user_id = ?';
  this.query({
    query: sqlQuery,
    data: [this.getInactiveID(), userID],
    cb: function(err, data) {
      if (cb) {
        cb(err, data);
      }
    }
  });
};

Authorization.prototype.getInactiveID = function() {
  return 0;
};

Authorization.prototype.beforeInsertUpdate = function() {
  var that = this;
  return [
    function(next) {
      that.invalidateToken(that.currentData.userID, next);
    }
  ];
};

module.exports = Authorization;
