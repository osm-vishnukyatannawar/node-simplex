api.validate = function(request, response, next) {
  var authObj = new Authorization();
  var cntrlObj = new Controller();
  authObj.checkValidity(request.params.token, function(err, data) {
    if (err) {
      cntrlObj.sendResponse(err, data, response);
    } else {
      next();
    }
  });
};

module.exports = api;
